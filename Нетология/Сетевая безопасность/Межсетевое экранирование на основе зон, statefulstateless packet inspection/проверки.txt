# === INSIDE → OUTSIDE ===
# Должно работать: ICMP
ping 10.10.10.2
ping 10.10.10.3

# Должно не работать: HTTP
curl http://10.10.10.2
curl http://10.10.10.3

# === INSIDE → DMZ ===
# Должно работать: HTTP
curl http://192.168.3.2

# Должно не работать: ICMP
ping 192.168.3.2

# === INSIDE → PRINTER ===
# Должно работать: ICMP
ping 192.168.2.2

# Должно не работать: HTTP или TCP
curl http://192.168.2.2

# === OUTSIDE → DMZ ===
# Должно работать: HTTP (инициация TCP/80)
curl http://192.168.3.2

# Должно не работать: к INSIDE и PRINTER
ping 192.168.1.2
ping 192.168.2.2
curl http://192.168.1.2
curl http://192.168.2.2

# === PRINTER → все зоны ===
# Должно не работать: любая инициированная связь
ping 192.168.1.2
ping 192.168.3.2
ping 10.10.10.2
curl http://192.168.1.2
curl http://192.168.3.2

# === DMZ → OUTSIDE ===
# Должно работать: ICMP
ping 10.10.10.2

# Должно не работать: к INSIDE и PRINTER
ping 192.168.1.2
ping 192.168.2.2
