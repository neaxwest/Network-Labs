=========================
 Cisco Password Cheatsheet
=========================

! --- 1. Пароль на консоль (Console) ---
enable
configure terminal
line console 0
password cisco
login
end
write memory

! Требует пароль при входе через консольный кабель


! --- 2. Пароль на VTY (Telnet/SSH) ---
enable
configure terminal
line vty 0 4
password cisco
login
end
write memory

! Требует пароль при удалённом входе


! --- 3. Enable Password (устаревший) ---
enable
configure terminal
enable password cisco
end
write memory

! Пароль хранится в открытом виде (не рекомендуется)


! --- 4. Enable Secret (рекомендуемый) ---
enable
configure terminal
enable secret MyStr0ngPass
end
write memory

! Перекрывает enable password
! Хранится в хэшированном виде (MD5)


! --- 5. Шифрование паролей в конфиге ---
enable
configure terminal
service password-encryption
end
write memory

! Простейшее шифрование (Type 7)
! Защищает от случайного просмотра в running-config


! --- 6. Локальные пользователи ---
enable
configure terminal
username admin privilege 15 secret Str0ngP@ss
end
write memory

! Создаёт пользователя "admin" с правами 15 (full admin)


! --- 7. Аутентификация VTY через локальных пользователей ---
enable
configure terminal
line vty 0 4
login local
transport input ssh
end
write memory

! Теперь можно подключаться по SSH
! Пример: ssh admin@192.168.1.1
